% Blakers-Massey: homotopy excision on a CW triad.
\begin{tikzpicture}[
  every node/.style={font=\small},
  arrow/.style={-{Stealth[length=2.5mm]}, thick},
]
  % CW triad: A, B, C = A cap B, X = A cup B
  \draw[blue!60!black, thick, fill=blue!10] (0, 0) ellipse [x radius=2.4cm, y radius=1.4cm];
  \draw[red!60!black, thick, fill=red!10]   (2.0, 0) ellipse [x radius=2.4cm, y radius=1.4cm];

  \node[blue!70!black] at (-2.0, 1.0) {$A$};
  \node[red!70!black]  at ( 4.0, 1.0) {$B$};
  \node[font=\footnotesize] at (1.0, 0) {$C = A \cap B$};

  \node[gray!50!black, font=\footnotesize] at (1.0, 1.7) {$X = A \cup B$};

  % Inclusion arrows + connectivity hypotheses
  \node[anchor=west, font=\small] at (5.5, 1.5)
    {$(A, C) \hookrightarrow (X, B)$ is $m$-connected};
  \node[anchor=west, font=\small] at (5.5, 0.8)
    {$(B, C) \hookrightarrow (X, A)$ is $n$-connected};

  % Theorem box
  \node[draw=teal!50!black, fill=teal!4, rounded corners=4pt, inner sep=4pt, align=left] at (5.5, -0.6)
    {{\bfseries Blakers-Massey:}\\
     $i_* : \pi_k(A, C) \to \pi_k(X, B)$\\
     iso for $k < m + n$, surjective at $m + n$.};

  % Freudenthal corollary
  \node[anchor=west, font=\footnotesize, gray!50!black] at (5.5, -2.3)
    {Freudenthal corollary: $\pi_k(X) \to \pi_{k+1}(\Sigma X)$ iso for $k < 2n - 1$};

  \begin{scope}[on background layer]
    \fill[teal!4, rounded corners=12pt] (-3.5, -3.0) rectangle (12.5, 2.5);
  \end{scope}

  \node[align=center] at (3.0, -3.6)
    {\itshape Homotopy version of excision: stable range where homotopy and homology agree.};
\end{tikzpicture}
