% Moser's trick: path of symplectic forms + flow of the Moser vector field.
\begin{tikzpicture}[
  every node/.style={font=\small},
  arrow/.style={-{Stealth[length=2.5mm]}, thick},
]
  % Time axis at the bottom
  \draw[->, thick, gray!50!black] (-0.5, 0) -- (8.5, 0) node[right, black] {$t$};
  \node at (0, -0.4) {$0$};
  \node at (8, -0.4) {$1$};

  % Path of symplectic forms (curve in "form space")
  \draw[blue!70!black, thick]
    (0, 2) .. controls (3, 3.2) and (5, 1.4) .. (8, 2.6);

  \node[blue!70!black, anchor=west] at (-0.5, 2.4) {$\omega_0$};
  \node[blue!70!black] at (8.4, 2.6) {$\omega_1$};

  % Some intermediate forms marked
  \filldraw[blue!70!black] (0, 2) circle (2.5pt);
  \filldraw[blue!70!black] (3, 2.7) circle (2pt) node[above, font=\footnotesize, black] {$\omega_t$};
  \filldraw[blue!70!black] (8, 2.6) circle (2.5pt);

  % Flow arrows beneath, indicating psi_t
  \draw[arrow, red!70!black] (0.4, 1.0) -- (8, 1.0);
  \node[red!70!black, anchor=west] at (-0.5, 1.0) {$\psi_t$};
  \node[red!70!black, font=\footnotesize] at (4, 0.55) {flow of $X_t$ with $\iota_{X_t}\omega_t = -\alpha_t$};

  % Conclusion box
  \node[draw=blue!60!black, fill=blue!4, rounded corners=4pt, inner sep=4pt, anchor=west] at (0.5, 4.3)
    {$\dfrac{d}{dt}(\psi_t^* \omega_t) \;=\; \psi_t^*(\mathcal{L}_{X_t}\omega_t + \dot\omega_t) \;=\; \psi_t^*(-d\alpha_t + d\alpha_t) \;=\; 0.$};

  \begin{scope}[on background layer]
    \fill[purple!4, rounded corners=12pt] (-1.5, -1.0) rectangle (10.0, 5.2);
  \end{scope}

  \node[align=center] at (4, -1.6)
    {\itshape Cartan-formula calculation: $\psi_1^*\omega_1 = \omega_0$, the desired symplectomorphism.};
\end{tikzpicture}
