% Cofibration: homotopy extension property as a retract diagram.
\begin{tikzpicture}[
  every node/.style={font=\small},
  arrow/.style={-{Stealth[length=2.5mm]}, thick},
]
  % Cylinder X x [0,1] sketch
  \draw[blue!50!black, thick] (0,0) ellipse [x radius=1.2cm, y radius=0.35cm];
  \draw[blue!50!black, thick] (0,3) ellipse [x radius=1.2cm, y radius=0.35cm];
  \draw[blue!50!black, thick] (-1.2,0) -- (-1.2,3);
  \draw[blue!50!black, thick] ( 1.2,0) -- ( 1.2,3);
  \node at (0,0) {$X \times \{0\}$};
  \node at (0,3) {$X \times \{1\}$};

  % Subspace A x [0,1] inside as a thin tube
  \filldraw[red!10] (-0.35,0.05) rectangle (0.35,2.95);
  \draw[red!70!black, thick] (-0.35,0) -- (-0.35,3);
  \draw[red!70!black, thick] ( 0.35,0) -- ( 0.35,3);
  \node[red!70!black] at (0,1.5) {$A \times [0,1]$};

  % Z target with maps
  \node (Z) at (5.5, 1.5) {$Z$};

  % f: X x {0} -> Z
  \draw[arrow, blue!60!black] (1.0, 0) to[bend right=15] node[below, font=\footnotesize] {$f$} (Z.south west);
  % H: A x [0,1] -> Z (homotopy on subspace)
  \draw[arrow, red!70!black] (0.4, 1.5) to[bend right=10] node[above, font=\footnotesize] {$H$} (Z.west);
  % tilde H: X x [0,1] -> Z (extension)
  \draw[arrow, dashed, gray!50!black] (1.2, 2.3) to[bend right=5] node[above, font=\footnotesize] {$\tilde H$} (Z.north west);

  \begin{scope}[on background layer]
    \fill[orange!4, rounded corners=12pt] (-2.7, -1.0) rectangle (7.4, 4.0);
  \end{scope}

  \node[align=center] at (2.5, -1.6)
    {\itshape The HEP: a homotopy on $A$ compatible with $f$ extends to all of $X$.};
\end{tikzpicture}
