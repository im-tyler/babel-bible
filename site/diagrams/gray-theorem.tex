% Gray's stability: smooth path of contact structures isotopic via psi_t.
\begin{tikzpicture}[
  every node/.style={font=\small},
  arrow/.style={-{Stealth[length=2.5mm]}, thick},
]
  % Time axis
  \draw[->, thick, gray!50!black] (-0.5, 0) -- (8.5, 0) node[right, black] {$t$};
  \node[font=\footnotesize] at (0, -0.4) {$0$};
  \node[font=\footnotesize] at (8, -0.4) {$1$};

  % Path of contact structures (curve in form space)
  \draw[teal!70!black, thick]
    (0, 2.4) .. controls (3, 3.4) and (5, 1.6) .. (8, 2.8);

  \node[teal!70!black, anchor=west] at (-0.5, 2.6) {$\xi_0$};
  \node[teal!70!black] at (8.4, 2.8) {$\xi_1$};

  \filldraw[teal!70!black] (0, 2.4) circle (2pt);
  \filldraw[teal!70!black] (4, 2.6) circle (2pt) node[above, black, font=\footnotesize] {$\xi_t$};
  \filldraw[teal!70!black] (8, 2.8) circle (2pt);

  % Isotopy psi_t
  \draw[arrow, red!70!black] (0.4, 1.0) -- (8, 1.0);
  \node[red!70!black, anchor=west] at (-0.5, 1.0) {$\psi_t$};
  \node[red!70!black, font=\footnotesize] at (4, 0.55) {$(\psi_t)_*\xi_0 = \xi_t$};

  % Boxed Moser-trick adaptation
  \node[draw=red!50!black, fill=red!4, rounded corners=4pt, inner sep=4pt, align=left] at (4.0, 4.4)
    {{\bfseries Gray:} solve $\iota_{X_t}d\alpha_t|_{\xi_t} = -\dot\alpha_t|_{\xi_t}$ for $X_t \in \xi_t$, integrate.};

  \begin{scope}[on background layer]
    \fill[teal!4, rounded corners=12pt] (-1.5, -1.0) rectangle (10.0, 5.4);
  \end{scope}

  \node[align=center] at (4, -1.6)
    {\itshape Closed manifold: contact structures in the same path-component are diffeomorphic.};
\end{tikzpicture}
