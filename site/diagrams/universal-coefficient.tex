% Universal coefficient theorem: split SES.
\begin{tikzpicture}[
  every node/.style={font=\small},
  arrow/.style={-{Stealth[length=2.5mm]}, thick},
]
  % Top: homology UCT
  \node at (-3, 4.6) {Homology UCT:};
  \node (Z1) at (-3, 4.0) {$0$};
  \node (A1) at ( 0, 4.0) {$H_n(X) \otimes G$};
  \node (B1) at ( 4, 4.0) {$H_n(X; G)$};
  \node (C1) at ( 8, 4.0) {$\mathrm{Tor}(H_{n-1}(X), G)$};
  \node (Z2) at (12, 4.0) {$0$};

  \draw[arrow] (Z1) -- (A1);
  \draw[arrow] (A1) -- (B1);
  \draw[arrow] (B1) -- (C1);
  \draw[arrow] (C1) -- (Z2);

  % Splitting indicated
  \draw[arrow, dashed, gray!50!black] (B1.south) to[bend right=30] node[below, font=\footnotesize] {splits} (A1.south);

  % Bottom: cohomology UCT
  \node at (-3, 1.8) {Cohomology UCT:};
  \node (Z3) at (-3, 1.2) {$0$};
  \node (A3) at ( 0, 1.2) {$\mathrm{Ext}(H_{n-1}(X), G)$};
  \node (B3) at ( 4, 1.2) {$H^n(X; G)$};
  \node (C3) at ( 8, 1.2) {$\mathrm{Hom}(H_n(X), G)$};
  \node (Z4) at (12, 1.2) {$0$};

  \draw[arrow] (Z3) -- (A3);
  \draw[arrow] (A3) -- (B3);
  \draw[arrow] (B3) -- (C3);
  \draw[arrow] (C3) -- (Z4);

  \draw[arrow, dashed, gray!50!black] (B3.south) to[bend right=30] node[below, font=\footnotesize] {splits} (A3.south);

  % Note
  \node[font=\small\itshape, anchor=west] at (-3.5, -0.4)
    {Field-coefficient case: $\mathrm{Tor}, \mathrm{Ext}$ vanish; $H_n(X; \mathbb{F}) \cong H_n(X) \otimes \mathbb{F}$,};
  \node[font=\small\itshape, anchor=west] at (-3.5, -0.9)
    {$H^n(X; \mathbb{F}) \cong \mathrm{Hom}(H_n(X), \mathbb{F})$.};

  \begin{scope}[on background layer]
    \fill[orange!4, rounded corners=12pt] (-4.5, -1.6) rectangle (13.4, 5.2);
  \end{scope}
\end{tikzpicture}
