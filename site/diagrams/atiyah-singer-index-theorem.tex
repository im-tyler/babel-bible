% Atiyah-Singer index theorem: the analytic-vs-topological square.
% Renders to site/public/img/atiyah-singer-index-theorem-placeholder.svg
% via scripts/render_diagram.py.
\begin{tikzpicture}[
  every node/.style={font=\small},
  arrow/.style={-{Stealth[length=3mm]}, thick},
]
  % Top row: analytic side
  \node (P)  at (0, 2)
    {$\bigl[\text{elliptic } P : \Gamma(E)\to\Gamma(F)\bigr] / \text{htpy}$};
  \node (Z1) at (8, 2) {$\mathbb{Z}$};

  % Bottom row: topological side
  \node (S)  at (0, 0)
    {$\bigl[\sigma(P)\bigr] \in K^0(T^*M)$};
  \node (Z2) at (8, 0) {$\mathbb{Z}$};

  % Horizontal arrows
  \draw[arrow] (P)  -- node[above] {analytic index}      node[below] {$\dim\ker - \dim\mathrm{coker}$} (Z1);
  \draw[arrow] (S)  -- node[above] {topological index}   node[below] {$\int_M \widehat{A}(M)\,\mathrm{ch}(\sigma)$} (Z2);

  % Vertical arrows
  \draw[arrow] (P)  -- node[left]  {principal symbol} (S);
  \draw[arrow,double,double distance=1.5pt,-] (Z1) -- node[right] {$=$} (Z2);

  % Background flourish to set off the diagram
  \begin{scope}[on background layer]
    \fill[blue!4, rounded corners=12pt]
      (-3.6, -1.0) rectangle (10.6, 3.0);
  \end{scope}

  % Caption
  \node at (3.5, -1.6)
    {\itshape index $=$ topology: the two routes to $\mathbb{Z}$ agree.};
\end{tikzpicture}
