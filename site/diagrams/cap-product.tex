% Cap product: front-face/back-face split of a singular n-simplex.
\begin{tikzpicture}[
  every node/.style={font=\small},
  vert/.style={fill=blue!75!black, circle, inner sep=2pt},
]
  % A singular 3-simplex (tetrahedron schematic) with front-face / back-face split for k=2.
  \node[vert, label=below:$v_0$]      (v0) at (0, 0) {};
  \node[vert, label=below:$v_1$]      (v1) at (3, 0) {};
  \node[vert, label=above:$v_2$]      (v2) at (1.6, 2.4) {};
  \node[vert, label=above right:$v_3$] (v3) at (3.8, 2.0) {};

  % Front face [v_0, v_1, v_2] (the "k-face" the cochain evaluates on)
  \filldraw[blue!50!black, fill=blue!12, opacity=0.6]
    (v0.center) -- (v1.center) -- (v2.center) -- cycle;

  % Back face [v_2, v_3] (the "(n-k)-face" left as a chain)
  \draw[red!75!black, very thick] (v2.center) -- (v3.center);

  % Tetrahedron edges (light)
  \draw[gray!40!black, thin] (v0.center) -- (v3.center);
  \draw[gray!40!black, thin] (v1.center) -- (v3.center);

  % Annotations
  \node[blue!60!black, font=\small] at (1.2, 1.0) {$\sigma|_{[v_0, v_1, v_2]}$};
  \node[red!75!black, font=\small] at (3.6, 2.4) {$\sigma|_{[v_2, v_3]}$};

  % Formula
  \node[anchor=west, font=\small] at (5.5, 1.4)
    {$\phi \frown \sigma \;=\; \phi(\sigma|_{[v_0, \ldots, v_k]}) \cdot \sigma|_{[v_k, \ldots, v_n]}$};
  \node[anchor=west, font=\footnotesize, gray!50!black] at (5.5, 0.7)
    {Front-face: cochain $\phi \in C^k$ evaluates here.};
  \node[anchor=west, font=\footnotesize, gray!50!black] at (5.5, 0.3)
    {Back-face: chain $\sigma|_{[v_k, \ldots, v_n]} \in C_{n-k}$ remains.};

  % Bottom: Leibniz rule
  \node[anchor=west, font=\footnotesize] at (5.5, -0.6)
    {$\partial(\phi \frown \sigma) = (-1)^k(\delta\phi \frown \sigma - \phi \frown \partial\sigma)$};

  \begin{scope}[on background layer]
    \fill[teal!4, rounded corners=12pt] (-1.0, -1.4) rectangle (12.5, 3.4);
  \end{scope}

  \node[align=center] at (5.0, -2.0)
    {\itshape Cap product makes $H_*(X)$ a graded $H^*(X)$-module via the Leibniz rule on chain level.};
\end{tikzpicture}
