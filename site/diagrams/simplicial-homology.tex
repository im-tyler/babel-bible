% Δ-complex structure on a torus: 2 triangles, 3 edges (a, b, c), 1 vertex.
\begin{tikzpicture}[
  every node/.style={font=\small},
  vert/.style={fill=blue!75!black, circle, inner sep=1.8pt},
]
  % Square representing the torus, with edge identifications a, b
  \draw[blue!50!black, thick] (0, 0) rectangle (5, 5);

  % Diagonal edge c
  \draw[red!70!black, thick] (0, 0) -- (5, 5);

  % Vertex (single — all four corners identified)
  \node[vert] at (0, 0) {};
  \node[vert] at (5, 0) {};
  \node[vert] at (0, 5) {};
  \node[vert] at (5, 5) {};

  % Edge labels with arrows
  \draw[-{Stealth[length=2mm]}, thick, blue!60!black] (1.8, 0)   -- (3.2, 0);
  \draw[-{Stealth[length=2mm]}, thick, blue!60!black] (1.8, 5)   -- (3.2, 5);
  \draw[-{Stealth[length=2mm]}, thick, teal!50!black] (0, 1.8)   -- (0, 3.2);
  \draw[-{Stealth[length=2mm]}, thick, teal!50!black] (5, 1.8)   -- (5, 3.2);
  \draw[-{Stealth[length=2mm]}, thick, red!70!black] (2.0, 2.0)  -- (3.0, 3.0);

  \node[blue!60!black] at (2.5, -0.5) {$a$};
  \node[blue!60!black] at (2.5,  5.5) {$a$};
  \node[teal!50!black] at (-0.5, 2.5) {$b$};
  \node[teal!50!black] at (5.5,  2.5) {$b$};
  \node[red!70!black]  at (3.6, 2.0) {$c$};

  % Face labels U, L
  \node at (3.5, 1.5) {$U$};
  \node at (1.5, 3.5) {$L$};

  % Chain complex annotation
  \node[anchor=west, font=\small] at (6.5, 4.2) {$C_0 = \mathbb{Z} \cdot v$};
  \node[anchor=west, font=\small] at (6.5, 3.4) {$C_1 = \mathbb{Z} \cdot a \oplus \mathbb{Z} \cdot b \oplus \mathbb{Z} \cdot c$};
  \node[anchor=west, font=\small] at (6.5, 2.6) {$C_2 = \mathbb{Z} \cdot U \oplus \mathbb{Z} \cdot L$};
  \node[anchor=west, font=\footnotesize, gray!50!black] at (6.5, 1.8)
    {$\partial U = b + c - a$};
  \node[anchor=west, font=\footnotesize, gray!50!black] at (6.5, 1.3)
    {$\partial L = a + c - b$};
  \node[anchor=west, font=\footnotesize, gray!50!black] at (6.5, 0.8)
    {$\partial(U - L) = -2(a - b)$};

  \begin{scope}[on background layer]
    \fill[teal!4, rounded corners=12pt] (-1.4, -1.4) rectangle (12.4, 6.4);
  \end{scope}

  \node at (5.5, -1.0)
    {\itshape Torus as a $\Delta$-complex: 1 vertex, 3 edges $a,b,c$, 2 triangles $U,L$. $H_*(\Sigma_1) = \mathbb{Z}, \mathbb{Z}^2, \mathbb{Z}, 0, \ldots$};
\end{tikzpicture}
