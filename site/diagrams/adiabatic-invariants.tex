  \draw[axis] (0, -0.5) -- (0, 5.5) node[above, black] {$p$};

  \draw[blue!75!black, very thick] (2.5, 2.5) ellipse [x radius=2.0cm, y radius=1.0cm];

  \node[teal!75!black, font=\footnotesize] at (1.4, 4.6) {$\lambda = \lambda_2$};

  % Action invariance box
  \node[draw=blue!50!black, fill=blue!4, rounded corners=4pt, inner sep=4pt] at (2.5, -1.0)
