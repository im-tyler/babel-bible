% Hamilton-Jacobi: action S(q,t) generating canonical transformation to H -> 0.
\begin{tikzpicture}[
  every node/.style={font=\small},
  arrow/.style={-{Stealth[length=2.5mm]}, thick},
]
  % Top: PDE box
  \node[draw=blue!50!black, fill=blue!4, rounded corners=4pt, inner sep=4pt] at (0, 3.5)
    {$\dfrac{\partial S}{\partial t} + H\!\left(q,\, \dfrac{\partial S}{\partial q},\, t\right) = 0$};

  % Cotangent bundle T*M
  \draw[teal!70!black, thick, fill=teal!8] (-3.5, 0) rectangle (3.5, 1.4);
  \node[teal!70!black] at (-3.0, 1.0) {$T^*M$};

  % Lagrangian solution L = graph of dS
  \draw[blue!75!black, very thick] (-3.4, 0.4) .. controls (-2.0, 1.0) and (1.0, 0.0) .. (3.4, 0.8);
  \node[blue!75!black, font=\footnotesize] at (3.0, 1.2) {$L = \{(q, dS_q)\}$};

  % Side: characteristics
  \node[anchor=west, font=\small] at (4.5, 1.0) {Characteristics:};
  \node[anchor=west, font=\small] at (4.5, 0.4) {$\dot q = \partial H/\partial p$};
  \node[anchor=west, font=\small] at (4.5, -0.2) {$\dot p = -\partial H/\partial q$};
  \node[anchor=west, font=\footnotesize, gray!50!black] at (4.5, -0.8) {(= Hamilton flow)};

  % Bottom annotations
  \node[anchor=west, font=\footnotesize] at (-3.5, -1.4)
    {Solutions of HJ are Lagrangian submanifolds of $T^*M$.};
  \node[anchor=west, font=\footnotesize] at (-3.5, -2.0)
    {WKB: HJ is the eikonal limit of Schrödinger as $\hbar \to 0$.};
  \node[anchor=west, font=\footnotesize] at (-3.5, -2.6)
    {Action-angle from HJ when $H$ is separable.};
