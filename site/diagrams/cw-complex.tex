% CW complex skeleton filtration with cellular pushout square.
\begin{tikzpicture}[
  every node/.style={font=\small},
  arrow/.style={-{Stealth[length=2.5mm]}, thick},
]
  % Skeleton chain bottom row
  \node (Xm1) at (-1.7, 0) {$\emptyset$};
  \node (X0)  at ( 0.4, 0) {$X^{(0)}$};
  \node (X1)  at ( 2.5, 0) {$X^{(1)}$};
  \node (X2)  at ( 4.6, 0) {$X^{(2)}$};
  \node (Xn)  at ( 6.7, 0) {$X^{(n)}$};
  \node       at ( 8.4, 0) {$\cdots$};
  \node (X)   at (10.0, 0) {$X$};

  \draw[arrow] (Xm1) -- node[above, font=\footnotesize] {$+\,0$-cells} (X0);
  \draw[arrow] (X0)  -- node[above, font=\footnotesize] {$+\,1$-cells} (X1);
  \draw[arrow] (X1)  -- node[above, font=\footnotesize] {$+\,2$-cells} (X2);
  \draw[arrow] (X2)  -- node[above, font=\footnotesize] {$\cdots$}     (Xn);
  \draw[arrow] (Xn)  -- ( 7.6, 0);
  \draw[arrow] ( 9.1, 0) -- node[above, font=\footnotesize] {$\mathrm{colim}$} (X);

  % Cellular pushout square above
  \node (Sup) at (3.5, 3.4) {$\bigsqcup_\alpha S^{n-1}$};
  \node (Dup) at (3.5, 2.0) {$\bigsqcup_\alpha D^n$};
  \node (Xnm1) at (6.4, 3.4) {$X^{(n-1)}$};
  \node (Xnnew) at (6.4, 2.0) {$X^{(n)}$};

  \draw[arrow] (Sup)  -- node[above] {$\Phi$ attaching} (Xnm1);
  \draw[arrow] (Sup)  -- node[left]  {$\hookrightarrow$} (Dup);
  \draw[arrow] (Xnm1) -- node[right] {$\hookrightarrow$} (Xnnew);
  \draw[arrow] (Dup)  -- node[below] {} (Xnnew);

  \node[font=\small\itshape] at (4.9, 2.7) {pushout};

  \begin{scope}[on background layer]
    \fill[purple!4, rounded corners=12pt] (-2.5, -0.9) rectangle (10.9, 4.2);
  \end{scope}

  \node[align=center] at (4.5, -1.6)
    {\itshape Skeleton filtration $X^{(n-1)} \to X^{(n)}$ as a pushout: $n$-cells attached along their boundaries.};
\end{tikzpicture}
