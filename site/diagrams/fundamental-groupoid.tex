% Fundamental groupoid: vertices = points, arrows = path-homotopy classes.
\begin{tikzpicture}[
  every node/.style={font=\small},
  vert/.style={fill=blue!70!black, circle, inner sep=2pt},
  arrow/.style={-{Stealth[length=2.5mm]}, thick},
  loop/.style={-{Stealth[length=2.5mm]}, thick},
]
  % A blob representing the space X
  \draw[gray!50!black, very thick, fill=gray!8] plot[smooth cycle]
    coordinates {(-3.5, 0) (-2, 2) (1, 2.4) (3, 1) (3.4, -1) (1.5, -2.2) (-1.5, -1.8) (-3, -1)};

  % Vertices (base-point set A)
  \node[vert, label={[font=\small]above:$a$}]   (a) at (-2.0, 0.6) {};
  \node[vert, label={[font=\small]above:$b$}]   (b) at ( 0.5, 1.3) {};
  \node[vert, label={[font=\small]right:$c$}]   (c) at ( 2.0,-0.4) {};
  \node[vert, label={[font=\small]below:$d$}]   (d) at (-0.6,-1.3) {};

  % Arrows: a few path-homotopy classes
  \draw[arrow, blue!75!black] (a) to[bend left=20]  node[above, font=\footnotesize] {$\gamma_1$} (b);
  \draw[arrow, blue!75!black] (b) to[bend left=15]  node[right, font=\footnotesize] {$\gamma_2$} (c);
  \draw[arrow, blue!75!black] (a) to[bend right=25] node[below, font=\footnotesize] {$\gamma_3$} (d);
  \draw[arrow, blue!75!black] (d) to[bend right=15] node[below, font=\footnotesize] {$\gamma_4$} (c);

  % A loop at b (a generator of pi_1(X, b))
  \draw[loop, red!70!black] (b) to[out=70, in=110, looseness=12]
    node[above, font=\footnotesize] {$\alpha$} (b);

  \node[anchor=west] at (3.6, 1.5) {$\mathrm{Aut}(b) = \pi_1(X, b)$};

  \begin{scope}[on background layer]
    \fill[blue!4, rounded corners=12pt] (-4.4, -2.7) rectangle (6.8, 3.0);
  \end{scope}

  \node at (1.0, -3.1)
    {\itshape Objects: points of $X$. Morphisms: path-homotopy classes. Composition: concatenation.};
\end{tikzpicture}
