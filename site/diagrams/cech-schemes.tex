% Čech complex on a scheme: cover by 3 affine opens, alternating-sum boundary.
\begin{tikzpicture}[
  every node/.style={font=\small},
  arrow/.style={-{Stealth[length=2.5mm]}, thick},
]
  % Three open sets U_0, U_1, U_2 covering X
  \draw[blue!60!black, thick, fill=blue!10] (-2.5, 0.5) circle (1.6cm);
  \draw[red!60!black, thick, fill=red!10]   (0.5, 1.6) circle (1.6cm);
  \draw[teal!60!black, thick, fill=teal!10] (-1.0, -1.6) circle (1.6cm);

  \node[blue!75!black] at (-3.5, 1.5) {$U_0$};
  \node[red!75!black]  at ( 1.5, 2.5) {$U_1$};
  \node[teal!75!black] at (-1.5, -2.7) {$U_2$};

  % Pairwise intersection points labeled
  \node[font=\footnotesize] at (-1.0, 1.1) {$U_{01}$};
  \node[font=\footnotesize] at (-1.7, -0.5) {$U_{02}$};
  \node[font=\footnotesize] at (0.0, 0.0) {$U_{12}$};

  % Triple intersection
  \filldraw[black] (-0.6, 0.2) circle (1.2pt);
  \node[font=\footnotesize] at (0.1, -0.5) {$U_{012}$};

  % Cech complex on the right
  \node[anchor=west] at (4.0, 2.3) {Čech complex:};
  \node[anchor=west, font=\small] at (4.0, 1.5)
    {$C^0 = \prod_i \mathcal{F}(U_i)$};
  \node[anchor=west, font=\small] at (4.0, 0.7)
    {$\to C^1 = \prod_{i<j} \mathcal{F}(U_{ij})$};
  \node[anchor=west, font=\small] at (4.0, -0.1)
    {$\to C^2 = \mathcal{F}(U_{012})$};

  \node[anchor=west, font=\footnotesize, gray!50!black] at (4.0, -1.0)
    {$\delta : (s_i) \mapsto (s_j|_{ij} - s_i|_{ij})$};

  \node[anchor=west, font=\small] at (4.0, -2.0)
    {$\check H^p(\mathcal{U}, \mathcal{F}) = \ker\delta^p / \mathrm{im}\,\delta^{p-1}$};

  \begin{scope}[on background layer]
    \fill[purple!4, rounded corners=12pt] (-4.6, -3.8) rectangle (12.0, 3.0);
  \end{scope}

  \node[align=center, font=\footnotesize] at (3.5, -3.4)
    {Cartan: separated $X$ + affine cover + quasi-coherent $\mathcal{F}$ $\Longrightarrow$ $\check H^* = H^*$.};
\end{tikzpicture}
