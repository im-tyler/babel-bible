% Legendre transform: convex function f and its dual f*.
\begin{tikzpicture}[
  every node/.style={font=\small},
  axis/.style={->, thick, gray!60!black},
]
  % Left side: convex f
  \draw[axis] (-4.5, 0) -- (0.0, 0) node[right, black] {$q$};
  \draw[axis] (-4.0, -0.4) -- (-4.0, 3.0) node[above, black] {$f(q)$};

  % Convex parabola
  \draw[blue!75!black, thick] (-3.5, 2.5) .. controls (-3.0, 0.5) and (-1.0, 0.4) .. (-0.5, 2.5);
  \node[blue!75!black] at (-2.0, 2.7) {$f$ convex};

  % Supporting line at slope p (tangent at some point)
  \draw[red!70!black, thick] (-3.4, 0.4) -- (-0.6, 2.6);
  \node[red!70!black, font=\footnotesize] at (-3.4, 0.6) {slope $p$};

  % Y-intercept = -f^*(p)
  \filldraw[red!70!black] (-4.0, 1.05) circle (2pt);
  \node[red!70!black, font=\footnotesize, anchor=east] at (-4.1, 1.05) {$-f^*(p)$};

  % Right side: dual f*
  \draw[axis] (1.5, 0) -- (6.0, 0) node[right, black] {$p$};
  \draw[axis] (2.0, -0.4) -- (2.0, 3.0) node[above, black] {$f^*(p)$};

  \draw[teal!75!black, thick] (2.5, 2.5) .. controls (3.0, 0.5) and (5.0, 0.4) .. (5.5, 2.5);
  \node[teal!75!black] at (4.0, 2.7) {$f^*$ convex (dual)};

  % Bottom annotation: Fenchel formula
  \node[draw=blue!50!black, fill=blue!4, rounded corners=4pt, inner sep=4pt] at (0.0, -1.4)
    {$f^*(p) = \sup_q\bigl(\langle p, q\rangle - f(q)\bigr) \qquad (f^*)^* = f$};

  \begin{scope}[on background layer]
    \fill[orange!4, rounded corners=12pt] (-5.5, -2.4) rectangle (7.0, 3.4);
  \end{scope}

  \node[align=center] at (0.5, -3.0)
    {\itshape Fibre Legendre transform $\mathbb{F}L : TQ \to T^*Q$ converts Lagrangian to Hamiltonian formalism.};
\end{tikzpicture}
