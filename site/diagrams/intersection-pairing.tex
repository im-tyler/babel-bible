% Intersection pairing on a surface: two curves meeting transversally.
\begin{tikzpicture}[
  every node/.style={font=\small},
  vert/.style={fill=black, circle, inner sep=1.5pt},
]
  % Surface as a curved patch
  \draw[gray!50!black, thick, fill=gray!8] plot[smooth cycle]
    coordinates {(-3.5, -2) (-3.5, 2) (3.5, 2) (3.5, -2)};
  \node[gray!60!black] at (-3.0, 1.7) {smooth surface $X$};

  % Two curves D and D' on X
  \draw[blue!75!black, very thick] (-3.2, 1.0) .. controls (-1.0, 0.4) and (1.0, -0.6) .. (3.2, -0.5);
  \draw[red!75!black, very thick]  (-3.0, -0.8) .. controls (-1.5, 0.2) and (1.5, 0.0) .. (3.2, 1.2);

  \node[blue!75!black] at (-3.4, 1.4) {$D$};
  \node[red!75!black]  at (-3.2, -1.2) {$D'$};

  % Transverse intersection points
  \node[vert] at (-1.3, 0.05) {};
  \node[vert] at ( 1.7, 0.30) {};

  % Annotations
  \node[anchor=west, font=\footnotesize] at (4.5, 1.0)
    {$D \cdot D' = $ \# transverse intersections};
  \node[anchor=west, font=\footnotesize] at (4.5, 0.4)
    {$= 2$ in this picture};

  % Equivalent definitions box
  \node[draw=blue!50!black, fill=blue!4, rounded corners=4pt, inner sep=4pt, align=left] at (1.0, -3.4)
    {{\bfseries Three equivalent definitions:}\\
     1. Geometric: count transverse intersections (after moving)\\
     2. Cohomological: $\chi(\mathcal{O}) - \chi(\mathcal{O}(-D)) - \chi(\mathcal{O}(-D')) + \chi(\mathcal{O}(-D-D'))$\\
     3. Cup-product on $H^2(X, \mathbb{Z})$ via the cycle-class map};

  \begin{scope}[on background layer]
    \fill[orange!4, rounded corners=12pt] (-5.0, -4.6) rectangle (10.0, 2.5);
  \end{scope}

  \node[align=center, font=\footnotesize] at (1.0, -5.2)
    {Self-intersection: $L^2 = 1$ on $\mathbb{P}^2$; $E^2 = -1$ for blow-up exceptional divisor; $K3$ rational curves $C^2 = -2$.};
\end{tikzpicture}
