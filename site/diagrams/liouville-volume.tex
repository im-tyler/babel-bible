  % Initial blob in phase space
  \draw[blue!60!black, very thick, fill=blue!10] (-4.0, 0) ellipse [x radius=1.0cm, y radius=0.7cm];
  \node[blue!60!black] at (-4.0, -1.2) {at $t = 0$};

  % Time-evolved blob (deformed but same area)
  \draw[red!60!black, very thick, fill=red!10] plot[smooth cycle]

  % Hamiltonian-flow arrow
  \draw[arrow, thick] (-2.7, 0) -- node[above] {Hamiltonian flow} (1.2, 0);

  % Conservation statement
  \node[draw=teal!50!black, fill=teal!4, rounded corners=4pt, inner sep=4pt] at (-0.5, 2.7)
    {$\mathcal{L}_{X_H}\, \omega^n = 0 \;\Longrightarrow\; \mathrm{vol}(\phi_t(B)) = \mathrm{vol}(B)$};

  % Side: Liouville equation for density
  \node[anchor=west, font=\small] at (5.5, 0.5) {Density evolution:};
  \node[anchor=west, font=\small] at (5.5, -0.2) {$\frac{\partial \rho}{\partial t} + \{\rho, H\} = 0$};
  \node[anchor=west, font=\footnotesize, gray!50!black] at (5.5, -0.9) {Equilibria: $\rho = f(H)$};

  \begin{scope}[on background layer]

    {\itshape Volume rigid + length flexible — the defining feature of symplectic geometry.};
