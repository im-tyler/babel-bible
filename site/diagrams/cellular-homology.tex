% Cellular chain complex of CP^n: zero in odd dimensions, Z in each even dimension.
\begin{tikzpicture}[
  every node/.style={font=\small},
  arrow/.style={-{Stealth[length=3mm]}, thick},
]
  % Header
  \node at (0, 4.6) {$C^{\mathrm{cell}}_*(\mathbb{CP}^n)$};

  % Chain complex bottom-to-top
  \node (Cn0) at (0, 0)   {$\mathbb{Z}$};      \node[gray!60!black, font=\footnotesize] at (-1.6, 0)   {$\deg 0$};
  \node (C1)  at (0, 0.8) {$0$};               \node[gray!60!black, font=\footnotesize] at (-1.6, 0.8) {$\deg 1$};
  \node (C2)  at (0, 1.6) {$\mathbb{Z}$};      \node[gray!60!black, font=\footnotesize] at (-1.6, 1.6) {$\deg 2$};
  \node (C3)  at (0, 2.4) {$0$};               \node[gray!60!black, font=\footnotesize] at (-1.6, 2.4) {$\deg 3$};
  \node (C4)  at (0, 3.2) {$\mathbb{Z}$};      \node[gray!60!black, font=\footnotesize] at (-1.6, 3.2) {$\deg 4$};
  \node       at (0, 3.9) {$\vdots$};

  % Boundary arrows (all zero — depicted with annotation)
  \draw[arrow, gray!60!black] (C2.south) -- node[right, font=\footnotesize] {$d = 0$} (C1.north);
  \draw[arrow, gray!60!black] (C1.south) -- node[right, font=\footnotesize] {$d = 0$} (Cn0.north);
  \draw[arrow, gray!60!black] (C3.south) -- node[right, font=\footnotesize] {$d = 0$} (C2.north);
  \draw[arrow, gray!60!black] (C4.south) -- node[right, font=\footnotesize] {$d = 0$} (C3.north);

  % Side annotation
  \node[anchor=west, font=\small] at (3.0, 2.6)
    {Cells in even dimensions only:};
  \node[anchor=west, font=\small] at (3.0, 2.0)
    {one $2k$-cell for $k = 0, 1, \ldots, n$.};
  \node[anchor=west, font=\small] at (3.0, 1.4)
    {Adjacent cells differ by 2;};
  \node[anchor=west, font=\small] at (3.0, 0.8)
    {no boundaries can fire.};
  \node[anchor=west, font=\small\itshape] at (3.0, 0.0)
    {Hence $H_k(\mathbb{CP}^n) = \mathbb{Z}$ for $k \in \{0, 2, \ldots, 2n\}$, 0 otherwise.};

  \begin{scope}[on background layer]
    \fill[purple!4, rounded corners=12pt] (-2.5, -1.0) rectangle (11.0, 5.0);
  \end{scope}
\end{tikzpicture}
