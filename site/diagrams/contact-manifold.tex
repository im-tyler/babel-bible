% Contact manifold: hyperplane field on R^3 = ker(dz - y dx). Standard "twisting" picture.
\begin{tikzpicture}[
  every node/.style={font=\small},
  axis/.style={->, thick, gray!60!black},
  plane/.style={fill=blue!20, draw=blue!60!black, opacity=0.6, thick},
]
  % Coordinate axes (3D illusion)
  \draw[axis] (0, 0) -- (5.0, 0) node[right, black] {$x$};
  \draw[axis] (0, 0) -- (0, 4.0) node[above, black] {$z$};
  \draw[axis] (0, 0) -- (-1.6, -1.2) node[below left, black] {$y$};

  % Contact planes ker(dz - y dx) at various (x, y, z) — they tilt as y varies
  % At y = -1: plane tilts down with slope -1 in dx-dz; at y = 0: horizontal; at y = 1: tilts up.
  \filldraw[plane] (1.0, 1.4) -- (2.0, 0.8) -- (2.0, 1.4) -- (1.0, 2.0) -- cycle;
  \filldraw[plane] (1.0, 2.4) -- (2.0, 2.4) -- (2.0, 3.0) -- (1.0, 3.0) -- cycle;
  \filldraw[plane] (3.0, 1.0) -- (4.0, 1.6) -- (4.0, 2.2) -- (3.0, 1.6) -- cycle;
  \filldraw[plane] (3.0, 2.4) -- (4.0, 2.4) -- (4.0, 3.0) -- (3.0, 3.0) -- cycle;

  % Annotations
  \node[font=\footnotesize, blue!60!black] at (1.55, 1.7) {$\xi_p$};
  \node[font=\footnotesize, blue!60!black] at (3.55, 1.5) {$\xi_p$};

  % Contact form formula
  \node[draw=blue!50!black, fill=blue!5, rounded corners=4pt, inner sep=4pt] at (8.0, 3.0)
    {$\alpha = dz - y\, dx$};
  \node[draw=teal!50!black, fill=teal!5, rounded corners=4pt, inner sep=4pt] at (8.0, 1.8)
    {$\xi = \ker \alpha$};
  \node[draw=red!50!black, fill=red!5, rounded corners=4pt, inner sep=4pt] at (8.0, 0.6)
    {$\alpha \wedge d\alpha = -dx \wedge dy \wedge dz \neq 0$};

  \begin{scope}[on background layer]
    \fill[orange!4, rounded corners=12pt] (-2.0, -1.6) rectangle (10.5, 4.5);
  \end{scope}

  \node[align=center] at (4.0, -2.4)
    {\itshape Standard contact structure on $\mathbb{R}^3$: hyperplane field $\xi$ tilts as $y$ varies, never integrating to a foliation.};
\end{tikzpicture}
